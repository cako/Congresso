leg_mats = []
for leg in legs:
    mats = 0
    for sen in leg.senadores:
        for mat in sen.materias:
            if mat.ano >= leg.anoini and mat.ano <= leg.anofim:
                mats += 1
    leg_mats.append(mats)
leg_mats_med = [ leg_mats[i]/len(legs[i].senadores) if len(legs[i].senadores) > 0 else 0 for i in range(len(legs)) ]

s_mats = pd.Series(leg_mats, index=[leg.anoini for leg in legs], name='Nmero de Matrias')
